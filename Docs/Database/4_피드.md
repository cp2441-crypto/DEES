---

```mermaid
erDiagram
    feeds {
        bigint id PK "고유 ID"
        bigint user_id FK "users.id (작성자)"
        enum feed_type "post / reels"
        text content "본문 (최대 2200자)"
        enum status "active / hidden / deleted"
        int like_count "좋아요 수 캐시"
        int comment_count "댓글 수 캐시"
        timestamp created_at "작성일"
        timestamp updated_at "수정일"
        timestamp deleted_at "삭제일"
    }

    feed_media {
        bigint id PK "고유 ID"
        bigint feed_id FK "feeds.id"
        enum media_type "image / video"
        varchar file_url "원본 파일 경로"
        varchar thumbnail_url "썸네일 경로"
        int file_size "파일 크기 (bytes)"
        int sort_order "정렬 순서"
        timestamp created_at "등록일"
    }

    feed_hashtags {
        bigint id PK "고유 ID"
        bigint feed_id FK "feeds.id"
        varchar hashtag "해시태그 (# 제외)"
    }

    feed_likes {
        bigint id PK "고유 ID"
        bigint feed_id FK "feeds.id"
        bigint user_id FK "users.id"
        timestamp created_at "좋아요일"
    }

    feed_comments {
        bigint id PK "고유 ID"
        bigint feed_id FK "feeds.id"
        bigint user_id FK "users.id (작성자)"
        bigint parent_id FK "feed_comments.id (대댓글 시 원댓글)"
        bigint mention_user_id FK "users.id (태그된 유저)"
        text content "댓글 내용 (1~500자)"
        enum status "active / hidden / deleted"
        timestamp created_at "작성일"
        timestamp updated_at "수정일"
    }

    feed_store_tags {
        bigint id PK "고유 ID"
        bigint feed_id FK "feeds.id"
        bigint store_id FK "stores.id (태그된 가게)"
    }

    feed_menu_tags {
        bigint id PK "고유 ID"
        bigint feed_id FK "feeds.id"
        bigint menu_id FK "menus.id (태그된 메뉴)"
    }

    feeds ||--o{ feed_media : "첨부 미디어"
    feeds ||--o{ feed_hashtags : "해시태그"
    feeds ||--o{ feed_likes : "좋아요"
    feeds ||--o{ feed_comments : "댓글"
    feeds ||--o{ feed_store_tags : "가게 태그"
    feeds ||--o{ feed_menu_tags : "메뉴 태그"
    feed_comments ||--o{ feed_comments : "대댓글"
```
