```mermaid
erDiagram
    stores {
        bigint id PK "고유 ID"
        bigint owner_id FK "users.id (점주)"
        varchar name "가게명"
        varchar address "주소"
        varchar phone "연락처"
        varchar business_hours "영업시간"
        text description "가게 소개"
        varchar sns_youtube "유튜브 링크"
        varchar sns_instagram "인스타그램 링크"
        varchar sns_tiktok "틱톡 링크"
        varchar naver_map_url "네이버 지도 Pin 링크"
        bigint category_id FK "대표 카테고리"
        enum status "active / paused / closed"
        int view_count "조회수"
        varchar delivery_company "택배사 (Phase 3)"
        int delivery_fee "기본 배송비 (Phase 3)"
        int delivery_fee_island "도서산간 추가 (Phase 3)"
        int free_delivery_amount "무료배송 기준 (Phase 3)"
        timestamp created_at "등록일"
        timestamp updated_at "수정일"
    }

    store_images {
        bigint id PK "고유 ID"
        bigint store_id FK "stores.id"
        varchar image_url "이미지 경로 (WEBP 변환)"
        varchar thumbnail_url "썸네일 경로 (360x360)"
        boolean is_primary "대표 이미지 여부"
        int sort_order "정렬 순서"
        timestamp created_at "등록일"
    }

    categories {
        bigint id PK "고유 ID"
        varchar code "카테고리 코드"
        varchar name "카테고리명 (케이크, 쿠키 등)"
        varchar icon "아이콘"
        boolean is_active "활성 여부"
        timestamp created_at "등록일"
    }

    menus {
        bigint id PK "고유 ID"
        bigint store_id FK "stores.id"
        varchar name "메뉴명"
        int price "가격 (원)"
        varchar description "한 줄 설명 (80자)"
        varchar image_url "메뉴 이미지 (WEBP)"
        varchar thumbnail_url "썸네일 (360x360)"
        bigint category_id FK "카테고리"
        int stock "재고 수량 (Phase 3)"
        boolean is_sold_out "품절 여부"
        boolean gift_wrapping "선물 포장 가능 (Phase 3)"
        enum season "spring/summer/autumn/winter (계절한정)"
        int order_count "주문 건수 (Phase 3)"
        int view_count "조회수"
        timestamp created_at "등록일"
        timestamp updated_at "수정일"
    }

    menu_options {
        bigint id PK "고유 ID"
        bigint menu_id FK "menus.id"
        varchar option_group "옵션 그룹 (사이즈, 맛 등)"
        varchar option_name "옵션명"
        int additional_price "추가 금액"
        boolean is_active "활성 여부"
        timestamp created_at "등록일"
    }

    tags {
        bigint id PK "고유 ID"
        varchar code "태그 코드 (TAG-VGN 등)"
        varchar name "태그명 (비건, 저당 등)"
        enum tag_type "health / allergy / attribute"
        varchar icon "노출 아이콘"
        boolean requires_verification "마스터 검증 필요 여부"
        boolean is_active "활성 여부"
    }

    menu_tags {
        bigint id PK "고유 ID"
        bigint menu_id FK "menus.id"
        bigint tag_id FK "tags.id"
        enum approval_status "pending / approved / rejected"
        bigint approved_by FK "승인 마스터 users.id"
        varchar evidence_url "증빙 서류 경로"
        timestamp approved_at "승인일"
        timestamp created_at "요청일"
    }

    allergens {
        bigint id PK "고유 ID"
        varchar code "알레르기 코드 (ALG-EGG 등)"
        varchar name "성분명 (난류, 우유 등)"
        varchar icon "경고 아이콘"
    }

    menu_allergens {
        bigint id PK "고유 ID"
        bigint menu_id FK "menus.id"
        bigint allergen_id FK "allergens.id"
    }

    stores ||--o{ store_images : "가게 이미지"
    stores ||--o{ menus : "메뉴 목록"
    stores }o--|| categories : "대표 카테고리"
    menus }o--|| categories : "메뉴 카테고리"
    menus ||--o{ menu_options : "옵션 (Phase 3)"
    menus ||--o{ menu_tags : "태그 연결"
    menus ||--o{ menu_allergens : "알레르기 연결"
    tags ||--o{ menu_tags : "태그 정보"
    allergens ||--o{ menu_allergens : "알레르기 정보"
```
