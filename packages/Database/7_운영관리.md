---
title: "DEES ERD ⑦ 운영·관리 도메인"
---

```mermaid
erDiagram
    reports {
        bigint id PK "고유 ID"
        bigint reporter_id FK "users.id (신고자)"
        enum report_type "store / feed / comment / review"
        bigint target_id "신고 대상 ID"
        enum reason "ad/abuse/fake/copyright/closed/inappropriate"
        text detail "상세 사유"
        enum status "pending / processing / resolved / dismissed"
        bigint processed_by FK "처리 마스터 users.id"
        varchar processor_note "처리 메모"
        boolean is_false_report "허위 신고 여부"
        timestamp processed_at "처리일"
        timestamp created_at "신고일"
    }

    sanctions {
        bigint id PK "고유 ID"
        bigint user_id FK "users.id (제재 대상)"
        bigint report_id FK "reports.id (관련 신고)"
        enum sanction_type "warning / suspend_7d / suspend_30d / permanent / exposure_7d / exposure_30d / contract_terminate"
        varchar reason "제재 사유"
        date start_date "제재 시작일"
        date end_date "제재 종료일"
        enum status "active / expired / revoked"
        bigint created_by FK "제재 마스터 users.id"
        timestamp created_at "등록일"
    }

    appeals {
        bigint id PK "고유 ID"
        bigint sanction_id FK "sanctions.id"
        bigint user_id FK "users.id (신청자)"
        text appeal_content "이의신청 내용"
        enum status "submitted / reviewing / accepted / rejected"
        text result_note "처리 결과 메모"
        bigint processed_by FK "처리자 users.id (슈퍼 마스터)"
        timestamp submitted_at "신청일"
        timestamp processed_at "처리일"
    }

    notifications {
        bigint id PK "고유 ID"
        bigint user_id FK "users.id (수신자)"
        enum channel "email / browser / sms"
        enum notification_type "sanction/dormant/follow_new/order/delivery/marketing"
        varchar title "알림 제목"
        text body "알림 내용"
        boolean is_read "읽음 여부"
        timestamp sent_at "발송일"
        timestamp read_at "읽은 일시"
    }

    notification_settings {
        bigint id PK "고유 ID"
        bigint user_id FK "users.id"
        varchar notification_type "알림 종류"
        boolean email_enabled "이메일 수신"
        boolean browser_enabled "브라우저 수신"
        boolean sms_enabled "SMS 수신 (Phase 3)"
        timestamp updated_at "수정일"
    }

    banners {
        bigint id PK "고유 ID"
        varchar title "배너 제목"
        varchar image_url "배너 이미지"
        varchar link_url "클릭 시 이동 URL"
        enum target_page "home / curation / store_list"
        int sort_order "노출 순서 (최대 5개)"
        date start_date "노출 시작일"
        date end_date "노출 종료일"
        boolean is_active "활성 여부"
        bigint created_by FK "등록 마스터 users.id"
        timestamp created_at "등록일"
    }

    popups {
        bigint id PK "고유 ID"
        varchar title "팝업 제목"
        varchar image_url "팝업 이미지"
        varchar link_url "클릭 시 이동 URL"
        enum close_option "close_today / close_permanent / close_only"
        date start_date "노출 시작일"
        date end_date "노출 종료일"
        boolean is_active "활성 여부"
        bigint created_by FK "등록 마스터 users.id"
        timestamp created_at "등록일"
    }

    admin_logs {
        bigint id PK "고유 ID"
        bigint master_id FK "users.id (마스터)"
        varchar action "수행 작업"
        varchar target_table "대상 테이블"
        bigint target_id "대상 레코드 ID"
        jsonb before_data "변경 전 데이터"
        jsonb after_data "변경 후 데이터"
        varchar ip_address "접속 IP"
        timestamp created_at "수행일시"
    }

    reports ||--o{ sanctions : "신고 → 제재"
    sanctions ||--o{ appeals : "제재 → 이의신청"
    users ||--o{ notifications : "알림 수신"
    users ||--o{ notification_settings : "알림 설정"
```
