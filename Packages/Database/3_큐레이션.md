---
title: "DEES ERD ③ 큐레이션 콘텐츠 도메인"
---

```mermaid
erDiagram
    brand_stories {
        bigint id PK "고유 ID"
        varchar title "제목"
        text body "본문 (리치 텍스트)"
        bigint created_by FK "작성 마스터 users.id"
        int view_count "조회수"
        timestamp created_at "등록일"
        timestamp updated_at "수정일"
    }

    brand_story_store_tags {
        bigint id PK "고유 ID"
        bigint brand_story_id FK "brand_stories.id"
        bigint store_id FK "stores.id (태그된 가게)"
    }

    dees_rank {
        bigint id PK "고유 ID"
        int rank_position "순위 (1~20)"
        bigint menu_id FK "menus.id"
        enum rank_method "manual / view_count / order_count"
        bigint assigned_by FK "지정 마스터 users.id (수동 시)"
        timestamp updated_at "갱신일"
    }

    famous_cafes {
        bigint id PK "고유 ID"
        bigint store_id FK "stores.id (연결된 가게)"
        varchar title "카페 제목"
        text description "카페 설명"
        varchar image_url "이미지"
        varchar area_sido "지역 - 시/도"
        int view_count "조회수"
        bigint created_by FK "등록 마스터 users.id"
        timestamp created_at "등록일"
        timestamp updated_at "수정일"
    }

    curation_likes {
        bigint id PK "고유 ID"
        bigint user_id FK "users.id"
        enum target_type "brand_story / famous_cafe / dees_rank"
        bigint target_id "대상 콘텐츠 ID"
        timestamp created_at "좋아요일"
    }

    curation_comments {
        bigint id PK "고유 ID"
        bigint user_id FK "users.id"
        enum target_type "brand_story / famous_cafe"
        bigint target_id "대상 콘텐츠 ID"
        text content "댓글 내용 (1~500자)"
        timestamp created_at "작성일"
        timestamp updated_at "수정일"
    }

    specials {
        bigint id PK "고유 ID"
        varchar title "기획전 제목"
        text description "설명"
        enum special_type "weekly / seasonal / exclusive"
        varchar banner_image_url "배너 이미지"
        date start_date "시작일"
        date end_date "종료일"
        int min_user_grade "접근 가능 최소 등급 (1~5)"
        bigint created_by FK "등록 마스터 users.id"
        boolean is_active "활성 여부"
        timestamp created_at "등록일"
        timestamp updated_at "수정일"
    }

    special_menus {
        bigint id PK "고유 ID"
        bigint special_id FK "specials.id"
        bigint menu_id FK "menus.id"
        int sort_order "노출 순서"
    }

    brand_stories ||--o{ brand_story_store_tags : "가게 태그"
    brand_stories ||--o{ curation_likes : "좋아요"
    brand_stories ||--o{ curation_comments : "댓글"
    famous_cafes ||--o{ curation_likes : "좋아요"
    famous_cafes ||--o{ curation_comments : "댓글"
    specials ||--o{ special_menus : "기획전 메뉴"
```
